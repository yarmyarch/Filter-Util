Run the demo in console:
FilterUtil.addFilter("value", function(value, offset) {
    alert(0);
    return value + offset;
}, 2);
FilterUtil.addFilter("value", function(value, offset) {
    alert(1);
    return value + offset * 2;
}, -1);
FilterUtil.addFilter("value", function(value, offset) {
    alert(2);
    return value + offset / 2;
}, 2);
FilterUtil.addFilter("value", function(value, offset) {
    alert(3);
    return value + offset;
}, 1);

var value = 2, offset = 1;
FilterUtil.applyFilter("value", value, offset);
Correct sequence should be 0->2->3->1.
