Run the demo in console:
FilterUtil.privateValue = 10;
FilterUtil.addFilter("value", function(value, offset) {
    alert(this.privateValue);
    return value + offset;
}, -1);
FilterUtil.addFilter("value", function(value, offset) {
    alert(1);
    return value + offset * 2;
}, 3);
FilterUtil.addFilter("value", function(value, offset) {
    alert(2);
    return value + offset / 2;
}, 1);
FilterUtil.addFilter("value", function(value, offset) {
    alert(3);
    return value + offset;
}, 2);

var value = 2, offset = 1;
FilterUtil.applyFilters("value", value, offset);
Correct sequence should be 10->2->3->1.
Enjoy.
